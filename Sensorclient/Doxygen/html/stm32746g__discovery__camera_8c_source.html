133 uint8_t BSP_CAMERA_Init(uint32_t Resolution)
134 {
135  DCMI_HandleTypeDef *phdcmi;
136  uint8_t status = CAMERA_ERROR;
137 
138  /* Get the DCMI handle structure */
139  phdcmi = &hDcmiHandler;
140 
141  /*** Configures the DCMI to interface with the camera module ***/
142  /* DCMI configuration */
143  phdcmi->Init.CaptureRate = DCMI_CR_ALL_FRAME;
144  phdcmi->Init.HSPolarity = DCMI_HSPOLARITY_LOW;
145  phdcmi->Init.SynchroMode = DCMI_SYNCHRO_HARDWARE;
146  phdcmi->Init.VSPolarity = DCMI_VSPOLARITY_HIGH;
147  phdcmi->Init.ExtendedDataMode = DCMI_EXTEND_DATA_8B;
148  phdcmi->Init.PCKPolarity = DCMI_PCKPOLARITY_RISING;
149  phdcmi->Instance = DCMI;
150 
151  /* Power up camera */
152  BSP_CAMERA_PwrUp();
153 
154  /* Read ID of Camera module via I2C */
155  if(ov9655_ReadID(CAMERA_I2C_ADDRESS) == OV9655_ID)
156  {
157  /* Initialize the camera driver structure */
158  camera_drv = &ov9655_drv;
159  CameraHwAddress = CAMERA_I2C_ADDRESS;
160 
161  /* DCMI Initialization */
162  BSP_CAMERA_MspInit(&hDcmiHandler, NULL);
163  HAL_DCMI_Init(phdcmi);
164 
165  /* Camera Module Initialization via I2C to the wanted 'Resolution' */
166  if (Resolution == CAMERA_R480x272)
167  { /* For 480x272 resolution, the OV9655 sensor is set to VGA resolution
168  * as OV9655 doesn't supports 480x272 resolution,
169  * then DCMI is configured to output a 480x272 cropped window */
170  camera_drv->Init(CameraHwAddress, CAMERA_R640x480);
171  HAL_DCMI_ConfigCROP(phdcmi, /* Crop in the middle of the VGA picture */
172  (CAMERA_VGA_RES_X - CAMERA_480x272_RES_X)/2,
173  (CAMERA_VGA_RES_Y - CAMERA_480x272_RES_Y)/2,
174  (CAMERA_480x272_RES_X * 2) - 1,
175  CAMERA_480x272_RES_Y - 1);
176  HAL_DCMI_EnableCROP(phdcmi);
177  }
178  else
179  {
180  camera_drv->Init(CameraHwAddress, Resolution);
181  HAL_DCMI_DisableCROP(phdcmi);
182  }
183 
184  CameraCurrentResolution = Resolution;
185 
186  /* Return CAMERA_OK status */
187  status = CAMERA_OK;
188  }
189  else
190  {
191  /* Return CAMERA_NOT_SUPPORTED status */
192  status = CAMERA_NOT_SUPPORTED;
193  }
194 
195  return status;
196 }

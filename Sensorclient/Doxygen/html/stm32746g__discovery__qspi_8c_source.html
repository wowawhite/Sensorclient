225 uint8_t BSP_QSPI_Write(uint8_t* pData, uint32_t WriteAddr, uint32_t Size)
226 {
227  QSPI_CommandTypeDef s_command;
228  uint32_t end_addr, current_size, current_addr;
229 
230  /* Calculation of the size between the write address and the end of the page */
231  current_size = N25Q128A_PAGE_SIZE - (WriteAddr % N25Q128A_PAGE_SIZE);
232 
233  /* Check if the size of the data is less than the remaining place in the page */
234  if (current_size > Size)
235  {
236  current_size = Size;
237  }
238 
239  /* Initialize the adress variables */
240  current_addr = WriteAddr;
241  end_addr = WriteAddr + Size;
242 
243  /* Initialize the program command */
244  s_command.InstructionMode = QSPI_INSTRUCTION_1_LINE;
245  s_command.Instruction = EXT_QUAD_IN_FAST_PROG_CMD;
246  s_command.AddressMode = QSPI_ADDRESS_4_LINES;
247  s_command.AddressSize = QSPI_ADDRESS_24_BITS;
248  s_command.AlternateByteMode = QSPI_ALTERNATE_BYTES_NONE;
249  s_command.DataMode = QSPI_DATA_4_LINES;
250  s_command.DummyCycles = 0;
251  s_command.DdrMode = QSPI_DDR_MODE_DISABLE;
252  s_command.DdrHoldHalfCycle = QSPI_DDR_HHC_ANALOG_DELAY;
253  s_command.SIOOMode = QSPI_SIOO_INST_EVERY_CMD;
254 
255  /* Perform the write page by page */
256  do
257  {
258  s_command.Address = current_addr;
259  s_command.NbData = current_size;
260 
261  /* Enable write operations */
262  if (QSPI_WriteEnable(&QSPIHandle) != QSPI_OK)
263  {
264  return QSPI_ERROR;
265  }
266 
267  /* Configure the command */
268  if (HAL_QSPI_Command(&QSPIHandle, &s_command, HAL_QPSI_TIMEOUT_DEFAULT_VALUE) != HAL_OK)
269  {
270  return QSPI_ERROR;
271  }
272 
273  /* Transmission of the data */
274  if (HAL_QSPI_Transmit(&QSPIHandle, pData, HAL_QPSI_TIMEOUT_DEFAULT_VALUE) != HAL_OK)
275  {
276  return QSPI_ERROR;
277  }
278 
279  /* Configure automatic polling mode to wait for end of program */
280  if (QSPI_AutoPollingMemReady(&QSPIHandle, HAL_QPSI_TIMEOUT_DEFAULT_VALUE) != QSPI_OK)
281  {
282  return QSPI_ERROR;
283  }
284 
285  /* Update the address and size variables for next page programming */
286  current_addr += current_size;
287  pData += current_size;
288  current_size = ((current_addr + N25Q128A_PAGE_SIZE) > end_addr) ? (end_addr - current_addr) : N25Q128A_PAGE_SIZE;
289  } while (current_addr < end_addr);
290 
291  return QSPI_OK;
292 }

88  int32_t LAN8742_Init(lan8742_Object_t *pObj)
89  {
90  uint32_t tickstart = 0, regvalue = 0, addr = 0;
91  int32_t status = LAN8742_STATUS_OK;
92 
93  if(pObj->Is_Initialized == 0)
94  {
95  if(pObj->IO.Init != 0)
96  {
97  /* GPIO and Clocks initialization */
98  pObj->IO.Init();
99  }
100 
101  /* for later check */
102  pObj->DevAddr = LAN8742_MAX_DEV_ADDR + 1;
103 
104  /* Get the device address from special mode register */
105  for(addr = 0; addr <= LAN8742_MAX_DEV_ADDR; addr ++)
106  {
107  if(pObj->IO.ReadReg(addr, LAN8742_SMR, &regvalue) < 0)
108  {
109  status = LAN8742_STATUS_READ_ERROR;
110  /* Can't read from this device address
111  continue with next address */
112  continue;
113  }
114 
115  if((regvalue & LAN8742_SMR_PHY_ADDR) == addr)
116  {
117  pObj->DevAddr = addr;
118  status = LAN8742_STATUS_OK;
119  break;
120  }
121  }
122 
123  if(pObj->DevAddr > LAN8742_MAX_DEV_ADDR)
124  {
125  status = LAN8742_STATUS_ADDRESS_ERROR;
126  }
127 
128  /* if device address is matched */
129  if(status == LAN8742_STATUS_OK)
130  {
131  /* set a software reset */
132  if(pObj->IO.WriteReg(pObj->DevAddr, LAN8742_BCR, LAN8742_BCR_SOFT_RESET) >= 0)
133  {
134  /* get software reset status */
135  if(pObj->IO.ReadReg(pObj->DevAddr, LAN8742_BCR, &regvalue) >= 0)
136  {
137  tickstart = pObj->IO.GetTick();
138 
139  /* wait until software reset is done or timeout occured */
140  while(regvalue & LAN8742_BCR_SOFT_RESET)
141  {
142  if((pObj->IO.GetTick() - tickstart) <= LAN8742_SW_RESET_TO)
143  {
144  if(pObj->IO.ReadReg(pObj->DevAddr, LAN8742_BCR, &regvalue) < 0)
145  {
146  status = LAN8742_STATUS_READ_ERROR;
147  break;
148  }
149  }
150  else
151  {
152  status = LAN8742_STATUS_RESET_TIMEOUT;
153  break;
154  }
155  }
156  }
157  else
158  {
159  status = LAN8742_STATUS_READ_ERROR;
160  }
161  }
162  else
163  {
164  status = LAN8742_STATUS_WRITE_ERROR;
165  }
166  }
167  }
168 
169  if(status == LAN8742_STATUS_OK)
170  {
171  tickstart = pObj->IO.GetTick();
172 
173  /* Wait for 2s to perform initialization */
174  while((pObj->IO.GetTick() - tickstart) <= LAN8742_INIT_TO)
175  {
176  }
177  pObj->Is_Initialized = 1;
178  }
179 
180  return status;
181  }

309 int32_t LAN8742_GetLinkState(lan8742_Object_t *pObj)
310 {
311  uint32_t readval = 0;
312 
313  /* Read Status register */
314  if(pObj->IO.ReadReg(pObj->DevAddr, LAN8742_BSR, &readval) < 0)
315  {
316  return LAN8742_STATUS_READ_ERROR;
317  }
318 
319  /* Read Status register again */
320  if(pObj->IO.ReadReg(pObj->DevAddr, LAN8742_BSR, &readval) < 0)
321  {
322  return LAN8742_STATUS_READ_ERROR;
323  }
324 
325  if((readval & LAN8742_BSR_LINK_STATUS) == 0)
326  {
327  /* Return Link Down status */
328  return LAN8742_STATUS_LINK_DOWN;
329  }
330 
331  /* Check Auto negotiaition */
332  if(pObj->IO.ReadReg(pObj->DevAddr, LAN8742_BCR, &readval) < 0)
333  {
334  return LAN8742_STATUS_READ_ERROR;
335  }
336 
337  if((readval & LAN8742_BCR_AUTONEGO_EN) != LAN8742_BCR_AUTONEGO_EN)
338  {
339  if(((readval & LAN8742_BCR_SPEED_SELECT) == LAN8742_BCR_SPEED_SELECT) && ((readval & LAN8742_BCR_DUPLEX_MODE) == LAN8742_BCR_DUPLEX_MODE))
340  {
341  return LAN8742_STATUS_100MBITS_FULLDUPLEX;
342  }
343  else if ((readval & LAN8742_BCR_SPEED_SELECT) == LAN8742_BCR_SPEED_SELECT)
344  {
345  return LAN8742_STATUS_100MBITS_HALFDUPLEX;
346  }
347  else if ((readval & LAN8742_BCR_DUPLEX_MODE) == LAN8742_BCR_DUPLEX_MODE)
348  {
349  return LAN8742_STATUS_10MBITS_FULLDUPLEX;
350  }
351  else
352  {
353  return LAN8742_STATUS_10MBITS_HALFDUPLEX;
354  }
355  }
356  else /* Auto Nego enabled */
357  {
358  if(pObj->IO.ReadReg(pObj->DevAddr, LAN8742_PHYSCSR, &readval) < 0)
359  {
360  return LAN8742_STATUS_READ_ERROR;
361  }
362 
363  /* Check if auto nego not done */
364  if((readval & LAN8742_PHYSCSR_AUTONEGO_DONE) == 0)
365  {
366  return LAN8742_STATUS_AUTONEGO_NOTDONE;
367  }
368 
369  if((readval & LAN8742_PHYSCSR_HCDSPEEDMASK) == LAN8742_PHYSCSR_100BTX_FD)
370  {
371  return LAN8742_STATUS_100MBITS_FULLDUPLEX;
372  }

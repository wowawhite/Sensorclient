206 void BSP_SDRAM_Initialization_sequence(uint32_t RefreshCount)
207 {
208  __IO uint32_t tmpmrd = 0;
209 
210  /* Step 1: Configure a clock configuration enable command */
211  Command.CommandMode = FMC_SDRAM_CMD_CLK_ENABLE;
212  Command.CommandTarget = FMC_SDRAM_CMD_TARGET_BANK1;
213  Command.AutoRefreshNumber = 1;
214  Command.ModeRegisterDefinition = 0;
215 
216  /* Send the command */
217  HAL_SDRAM_SendCommand(&sdramHandle, &Command, SDRAM_TIMEOUT);
218 
219  /* Step 2: Insert 100 us minimum delay */
220  /* Inserted delay is equal to 1 ms due to systick time base unit (ms) */
221  HAL_Delay(1);
222 
223  /* Step 3: Configure a PALL (precharge all) command */
224  Command.CommandMode = FMC_SDRAM_CMD_PALL;
225  Command.CommandTarget = FMC_SDRAM_CMD_TARGET_BANK1;
226  Command.AutoRefreshNumber = 1;
227  Command.ModeRegisterDefinition = 0;
228 
229  /* Send the command */
230  HAL_SDRAM_SendCommand(&sdramHandle, &Command, SDRAM_TIMEOUT);
231 
232  /* Step 4: Configure an Auto Refresh command */
233  Command.CommandMode = FMC_SDRAM_CMD_AUTOREFRESH_MODE;
234  Command.CommandTarget = FMC_SDRAM_CMD_TARGET_BANK1;
235  Command.AutoRefreshNumber = 8;
236  Command.ModeRegisterDefinition = 0;
237 
238  /* Send the command */
239  HAL_SDRAM_SendCommand(&sdramHandle, &Command, SDRAM_TIMEOUT);
240 
241  /* Step 5: Program the external memory mode register */
242  tmpmrd = (uint32_t)SDRAM_MODEREG_BURST_LENGTH_1 |\
243  SDRAM_MODEREG_BURST_TYPE_SEQUENTIAL |\
244  SDRAM_MODEREG_CAS_LATENCY_2 |\
245  SDRAM_MODEREG_OPERATING_MODE_STANDARD |\
246  SDRAM_MODEREG_WRITEBURST_MODE_SINGLE;
247 
248  Command.CommandMode = FMC_SDRAM_CMD_LOAD_MODE;
249  Command.CommandTarget = FMC_SDRAM_CMD_TARGET_BANK1;
250  Command.AutoRefreshNumber = 1;
251  Command.ModeRegisterDefinition = tmpmrd;
252 
253  /* Send the command */
254  HAL_SDRAM_SendCommand(&sdramHandle, &Command, SDRAM_TIMEOUT);
255 
256  /* Step 6: Set the refresh rate counter */
257  /* Set the device refresh rate */
258  HAL_SDRAM_ProgramRefreshRate(&sdramHandle, RefreshCount);
259 }

510 #ifndef SYS_ARCH_INC
511 #define SYS_ARCH_INC(var, val) do { \
512  SYS_ARCH_DECL_PROTECT(old_level); \
513  SYS_ARCH_PROTECT(old_level); \
514  var += val; \
515  SYS_ARCH_UNPROTECT(old_level); \
516  } while(0)
517 #endif /* SYS_ARCH_INC */

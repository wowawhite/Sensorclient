806 
807  deltax = ABS(x2 - x1); /* The difference between the x's */
808  deltay = ABS(y2 - y1); /* The difference between the y's */
809  x = x1; /* Start x off at the first pixel */
810  y = y1; /* Start y off at the first pixel */
811 
812  if (x2 >= x1) /* The x-values are increasing */
813  {
814  xinc1 = 1;
815  xinc2 = 1;
816  }
817  else /* The x-values are decreasing */
818  {

821  }
822

824  {

826  yinc2 = 1;
827  }
828  else /* The y-values are decreasing */
829  {
830  yinc1 = -1;
831  yinc2 = -1;
832  }
833 
834  if (deltax >= deltay) /* There is at least one x-value for every y-value */
835  {
836  xinc1 = 0; /* Don't change the x when numerator >= denominator */
837  yinc2 = 0; /* Don't change the y for every iteration */
838  den = deltax;
839  num = deltax / 2;
840  num_add = deltay;
841  num_pixels = deltax; /* There are more x-values than y-values */
842  }
843  else /* There is at least one y-value for every x-value */
844  {
845  xinc2 = 0; /* Don't change the x for every iteration */
846  yinc1 = 0; /* Don't change the y when numerator >= denominator */
847  den = deltay;
848  num = deltay / 2;
849  num_add = deltax;

851  }

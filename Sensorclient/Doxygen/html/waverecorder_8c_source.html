67 /*
68  A double MEMS microphone MP45DT02 mounted on STM32746G-DISCOVERY is connected
69  to the WM8994 audio codec. The SAI is configured in master
70  receiver mode. In this mode, the SAI provides the clock to the WM8994. The
71  WM8994 generates a clock to MEMS and acquires the data (Audio samples) from the MEMS
72  microphone in PDM format. WM8994 performs PDM to PCM format conversion before
73  sending samples to STM32.
74 
75  Data acquisition is performed in 16-bit PCM format and using SAI DMA mode.
76 
77  DMA is configured in circular mode
78 
79  In order to avoid data-loss, a 128 bytes buffer is used (BufferCtl.pdm_buff):
80  - When a DMA half transfer is detected using the call back BSP_AUDIO_IN_HalfTransfer_CallBack()
81  PCM frame is saved in RecBuf.
82  - The samples are then stored in USB buffer.
83  - These two steps are repeated when the DMA Transfer complete interrupt is detected
84  - When half of internal USB buffer is reach, an evacuation though USB is done.
85 
86  To avoid data-loss:
87  - IT ISR priority must be set at a higher priority than USB, this priority
88  order must be respected when managing other interrupts;
89 */

495 static uint32_t WavProcess_EncInit(uint32_t Freq, uint8_t *pHeader)
496 {
497  /* Initialize the encoder structure */
498  WaveFormat.SampleRate = Freq; /* Audio sampling frequency */
499  WaveFormat.NbrChannels = 2; /* Number of channels: 1:Mono or 2:Stereo */
500  WaveFormat.BitPerSample = 16; /* Number of bits per sample (16, 24 or 32) */
501  WaveFormat.FileSize = 0x001D4C00; /* Total length of useful audio data (payload) */
502  WaveFormat.SubChunk1Size = 44; /* The file header chunk size */
503  WaveFormat.ByteRate = (WaveFormat.SampleRate * \
504  (WaveFormat.BitPerSample/8) * \
505  WaveFormat.NbrChannels); /* Number of bytes per second (sample rate * block align) */
506  WaveFormat.BlockAlign = WaveFormat.NbrChannels * \
507  (WaveFormat.BitPerSample/8); /* channels * bits/sample / 8 */
508 
509  /* Parse the wav file header and extract required information */
510  if(WavProcess_HeaderInit(pHeader, &WaveFormat))
511  {
512  return 1;
513  }
514  return 0;
515 }

523 static uint32_t WavProcess_HeaderInit(uint8_t* pHeader, WAVE_FormatTypeDef* pWaveFormatStruct)
524 {
525  /* Write chunkID, must be 'RIFF' ------------------------------------------*/
526  pHeader[0] = 'R';
527  pHeader[1] = 'I';
528  pHeader[2] = 'F';
529  pHeader[3] = 'F';
530 
531  /* Write the file length ---------------------------------------------------*/
532  /* The sampling time: this value will be written back at the end of the
533  recording operation. Example: 661500 Btyes = 0x000A17FC, byte[7]=0x00, byte[4]=0xFC */
534  pHeader[4] = 0x00;
535  pHeader[5] = 0x4C;
536  pHeader[6] = 0x1D;
537  pHeader[7] = 0x00;
538  /* Write the file format, must be 'WAVE' -----------------------------------*/
539  pHeader[8] = 'W';
540  pHeader[9] = 'A';
541  pHeader[10] = 'V';
542  pHeader[11] = 'E';
543 
544  /* Write the format chunk, must be'fmt ' -----------------------------------*/
545  pHeader[12] = 'f';
546  pHeader[13] = 'm';
547  pHeader[14] = 't';
548  pHeader[15] = ' ';
549 
550  /* Write the length of the 'fmt' data, must be 0x10 ------------------------*/
551  pHeader[16] = 0x10;
552  pHeader[17] = 0x00;
553  pHeader[18] = 0x00;
554  pHeader[19] = 0x00;
555 
556  /* Write the audio format, must be 0x01 (PCM) ------------------------------*/
557  pHeader[20] = 0x01;
558  pHeader[21] = 0x00;
559 
560  /* Write the number of channels, ie. 0x01 (Mono) ---------------------------*/
561  pHeader[22] = pWaveFormatStruct->NbrChannels;
562  pHeader[23] = 0x00;
563 
564  /* Write the Sample Rate in Hz ---------------------------------------------*/
565  /* Write Little Endian ie. 8000 = 0x00001F40 => byte[24]=0x40, byte[27]=0x00*/
566  pHeader[24] = (uint8_t)((pWaveFormatStruct->SampleRate & 0xFF));
567  pHeader[25] = (uint8_t)((pWaveFormatStruct->SampleRate >> 8) & 0xFF);
568  pHeader[26] = (uint8_t)((pWaveFormatStruct->SampleRate >> 16) & 0xFF);
569  pHeader[27] = (uint8_t)((pWaveFormatStruct->SampleRate >> 24) & 0xFF);
570 
571  /* Write the Byte Rate -----------------------------------------------------*/

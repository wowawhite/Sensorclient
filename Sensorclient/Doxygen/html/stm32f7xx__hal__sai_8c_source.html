901  {
902  return HAL_ERROR;
903  }
904

906  {
907  /* Process Locked */
908  __HAL_LOCK(hsai);
909 
910  hsai->XferSize = Size;
911  hsai->XferCount = Size;
912  hsai->pBuffPtr = pData;
913  hsai->State = HAL_SAI_STATE_BUSY_TX;
914  hsai->ErrorCode = HAL_SAI_ERROR_NONE;
915

929  {

931  {
932  hsai->Instance->DR = (*hsai->pBuffPtr++);
933  }

935  {

937  hsai->pBuffPtr += 2;
938  }

942  hsai->pBuffPtr += 4;

945  }

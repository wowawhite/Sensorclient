234 
235  if(tsOrientation == TS_SWAP_NONE)
236  {
237  x[index] = brute_x[index];
238  y[index] = brute_y[index];
239  }
240 
241  if(tsOrientation & TS_SWAP_X)
242  {
243  x[index] = 4096 - brute_x[index];
244  }
245 
246  if(tsOrientation & TS_SWAP_Y)
247  {
248  y[index] = 4096 - brute_y[index];
249  }
250

253  y[index] = brute_x[index];
254  x[index] = brute_y[index];
255  }
256 
257  x_diff = x[index] > _x[index]? (x[index] - _x[index]): (_x[index] - x[index]);
258  y_diff = y[index] > _y[index]? (y[index] - _y[index]): (_y[index] - y[index]);
259 
260  if ((x_diff + y_diff) > 5)
261  {
262  _x[index] = x[index];
263  _y[index] = y[index];
264  }
265 
266  if(I2cAddress == FT5336_I2C_SLAVE_ADDRESS)
267  {

269  TS_State->touchY[index] = y[index];
270  }

274  TS_State->touchX[index] = (tsXBoundary * _x[index]) >> 12;
275  TS_State->touchY[index] = (tsYBoundary * _y[index]) >> 12;

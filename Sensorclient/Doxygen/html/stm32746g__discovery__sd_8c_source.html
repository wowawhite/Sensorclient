132 uint8_t BSP_SD_Init(void)
133 {
134  uint8_t sd_state = MSD_OK;
135 
136  /* uSD device interface configuration */
137  uSdHandle.Instance = SDMMC1;
138 
139  uSdHandle.Init.ClockEdge = SDMMC_CLOCK_EDGE_RISING;
140  uSdHandle.Init.ClockBypass = SDMMC_CLOCK_BYPASS_DISABLE;
141  uSdHandle.Init.ClockPowerSave = SDMMC_CLOCK_POWER_SAVE_DISABLE;
142  uSdHandle.Init.BusWide = SDMMC_BUS_WIDE_1B;
143  uSdHandle.Init.HardwareFlowControl = SDMMC_HARDWARE_FLOW_CONTROL_DISABLE;
144  uSdHandle.Init.ClockDiv = SDMMC_TRANSFER_CLK_DIV;
145 
146  /* Msp SD Detect pin initialization */
147  BSP_SD_Detect_MspInit(&uSdHandle, NULL);
148  if(BSP_SD_IsDetected() != SD_PRESENT) /* Check if SD card is present */
149  {
150  return MSD_ERROR_SD_NOT_PRESENT;
151  }
152 
153  /* Msp SD initialization */
154  BSP_SD_MspInit(&uSdHandle, NULL);
155 
156  /* HAL SD initialization */
157  if(HAL_SD_Init(&uSdHandle) != HAL_OK)
158  {
159  sd_state = MSD_ERROR;
160  }
161 
162  /* Configure SD Bus width */
163  if(sd_state == MSD_OK)
164  {
165  /* Enable wide operation */
166  if(HAL_SD_ConfigWideBusOperation(&uSdHandle, SDMMC_BUS_WIDE_4B) != HAL_OK)
167  {
168  sd_state = MSD_ERROR;
169  }
170  else
171  {
172  sd_state = MSD_OK;
173  }
174  }
175 
176  return sd_state;
177 }
